   public void paint(Graphics g)
   {
      //Set the color to yellow for the star
      g.setColor(Color.YELLOW);
      
      // Coordinates for the center of the star and the size (radius)
      int centerX = 200;
      int centerY = 200;
      int radius = 100;
      
      // Draw the five-pointed star using the fillPolygon method
      drawStar(g, centerX, centerY, radius, 5);
   }
   
   // Method to draw a five-pointed star
   private void drawStar(Graphics g, int centerX, int centerY, int radius, int points)
   {
      int[] xPoints = new int[2 * points];
      int[] yPoints = new int[2 * points];
      
      double angle = Math.PI / points; // Half the angle between two points
      
      // Calculate the points of the star
      for (int i = 0; i < 2 * points; i++)
      {
         // Alternate between outer and inner points
         int r = (i % 2 == 0) ? radius : radius / 2;
         double theta = i * angle; // Angle for each point
         xPoints[i] = (int) (centerX + r * Math.cos(theta));
         yPoints[i] = (int) (centerY - r * Math.sin(theta)); // Y-axis is inverted in Java's Graphics system
         
      }
      
      // Draw the star using fillPolygon
      g.fillPolygon(xPoints, yPoints, 2 * points);
   }
   
   public static void main(String[] args)
   {
      // Create the JFrame
      JFrame frame = new JFrame("Five-Pointed Star");
      Canvas canvas = new StarDrawing();
      
      // Set the size of the canvas and JFrame
      canvas.setSize(400, 400);
      frame.setSize(400, 400);
      frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
      
      // Add the canvas to the JFrame
      frame.add(canvas);
      
      // Set the frame visible
      frame.setVisible(true);
   }
}

   {
      JFrame frame = new JFrame("Chinese National Flag");
      ChineseFlag panel = new ChineseFlag();

      frame.setSize(FLAG_WIDTH, FLAG_HEIGHT);

   }

   @Override
   protected void paintComponent(Graphics g)
   {
      super.paintComponent(g);
      // Get the graphics object
      Graphics2D g2d = (Graphics2D) g;
      // Anti-aliasing for smoother lines
      g2d.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
      // Set the background to red (the flag's background color)
      g2d.setColor(new Color(222, 41, 19));
      // Red color of the flag
      g2d.fillRect(0, 0, FLAG_WIDTH, FLAG_HEIGHT);
      // Draw the large star

      drawStar(g2d, 100, 80, 50, 20, 5);
      // Draw the smaller stars
      drawSmallStars(g2d);
   }
   // Method to draw a star at a given position
   private void drawStar(Graphics2D g2d, int x, int y, int radius1, int radius2, int numPoints)
   {
      // The angle between each point on the star
      double angle = Math.PI / numPoints;
      // Create a polygon representing the star
      GeneralPath path = new GeneralPath();
      // First point of the star
      path.moveTo(x + Math.cos(0) * radius1, y - Math.sin(0) * radius1);
      // Alternate between outer and inner points
      for (int i = 1; i < 2 * numPoints; i++)
      {
         double currentAngle = i * angle;
         double currentRadius = (i % 2 == 0) ? radius1 : radius2;
         path.lineTo(x + Math.cos(currentAngle) * currentRadius, y - Math.sin(currentAngle) * currentRadius);
      }
      path.closePath();
      g2d.fill(path);
   }
   // Draw the smaller stars on the flag
   private void drawSmallStars(Graphics2D g2d)
   {
      int[][] positions =
      {
         {
            200, 50
         }
         ,

         }
         // Relative positions for the smaller stars
         
      };
      for (int[] position : positions)
      {
         drawStar(g2d, position[0], position[1], 20, 8, 5);
      }
   }

   private int flagWidth;
   private int flagHeight;
   
   public FlagCanvas(int width)
   {
      flagWidth = width;
      flagHeight = (int)(width * 2.0 / 3.0);
   }
   
   @Override
   public void paint(Graphics g)
   {
      // Draw the red rectangle background of the flag
      g.setColor(Color.RED);
      g.fillRect(0, 0, flagWidth, flagHeight);
      
      // Calculate the dimensions for the moon and star
      int moonCenterX = (int)(flagWidth * 0.2); // Position of the moon
      int moonCenterY = (int)(flagHeight * 0.5); // Moon centered vertically
      int moonRadius = flagHeight / 6; // Radius of the moon
      int starCenterX = (int)(flagWidth * 0.25); // Position of the star
      int starCenterY = moonCenterY - moonRadius / 2; // Star placed above the moon
      int starRadius = moonRadius / 2; // Radius of the star
      
      // Paint the white crescent moon
      g.setColor(Color.WHITE);
      g.fillArc(moonCenterX - moonRadius, moonCenterY - moonRadius, 2 * moonRadius, 2 * moonRadius, 90, 180); // White outer arc
      g.setColor(Color.RED);
      g.fillArc(moonCenterX - moonRadius, moonCenterY - moonRadius, 2 * moonRadius, 2 * moonRadius, 90, 180); // Red inner arc
      g.setColor(Color.WHITE);
      g.fillArc(moonCenterX - moonRadius / 2, moonCenterY - moonRadius / 2, moonRadius, moonRadius, 90, 180); // White inner arc
      // Paint the white five-pointed star
      g.setColor(Color.WHITE);
      int[][] starVertices = vertices(starCenterX, starCenterY, starRadius, 0);
      g.fillPolygon(starVertices[0], starVertices[1], 10);
   }
   
   // Method to calculate the vertices of a star
   int[][] vertices(double cx, double cy, double r, double angle)
   {
      double outerRadius = 1.0;
      double innerRadius = outerRadius * Math.sin(Math.PI / 10) / Math.sin(3 * Math.PI / 10); // Calculate inner radius based on trigonometry
      
      double angleIncrement = (Math.PI / 5);
      double startAngle = angle;
      int[][] d = new int[2][10];
      for (int i = 0; i < 10; i++)
      {
         double currentAngle = startAngle + i * angleIncrement;
         double radius = (i % 2 == 0) ? outerRadius : innerRadius;
         double x = r * radius * Math.cos(currentAngle) + cx;
         double y = r * radius * Math.sin(currentAngle) + cy;
         d[0][i] = (int) Math.round(x);
         d[1][i] = (int) Math.round(y);
      }
      return d;
   }

      frame.setSize(canvas.flagWidth, canvas.flagHeight);
      frame.getContentPane().add(canvas);
      frame.setVisible(true);
   }

   private void saveData()
   {
      StringBuilder stringBuilder = new StringBuilder();
      for (int row = 0; row < totalRows; row++)
      {
         for (int col = 0; col < getComponentCount(); col++)
         {
            Component comp = findComponentAt(row, col);

               {
                  value = ((JLabel) comp).getText();
               }
               else if (comp instanceof JTextField)
               {
                  value = ((JTextField) comp).getText();
               }
               else if (comp instanceof JComboBox)
               {
                  value = ((JComboBox<?>) comp).getSelectedItem().toString();
               }
               stringBuilder.append(value);
               if (col < 9 - 1)
               {
                  // Assuming 9 columns
                  stringBuilder.append("\t"); // Using tab as delimiter
                  
               }
            }
         }
         stringBuilder.append("\n");
      }

      }
      catch (IOException e)
      {
         JOptionPane.showMessageDialog(this, "Error saving data: "+ e.getMessage());
      }
   }

         }
      }
      return null;
   }

   public SimpleHTTPServer(int port)

   }
   public void start()
   {
      try
      {
         serverSocket = new ServerSocket(port);
         System.out.println("Server started on port "+ port);
         while (true)
         {
            Socket clientSocket = serverSocket.accept();
            Thread thread =new Thread(() ->handleClient(clientSocket));
            thread.start();
         }
      }
      catch (IOException e)
      {
         e.printStackTrace();
      }
   }
   private void handleClient(Socket clientSocket)
   {
      try
      {
         BufferedReader in =new BufferedReader(new
         InputStreamReader(clientSocket.getInputStream()));
         DataOutputStream out = new DataOutputStream(clientSocket.getOutputStream());
         // Read the request headers
         Map headers = readHeaders(in);
         System.out.println(headers);
         // Check if it's a multipart/form-data request
         //Send resDonse
         String response="HTTP/1.1 200 0K\r\n"
         +"Content-Type: text/plain\r\n"
         +"\r\n"
         +"request received";
         out.writeBytes(response);
         in.close();
         out.close();
         clientSocket.close();
      }
      catch (IOException e)
      {
         e.printStackTrace();
      }
   }
   private Map readHeaders(BufferedReader in)throws IOException
   {
      Map headers = new HashMap<>();
      String line;
      while(!(line = in.readLine()).isEmpty())
      {
         String[] parts = line.split(":", 2);
         if(parts.length ==2)
         {
            headers.put(parts[0].trim(), parts[1].trim());
         }
      }
      return headers;
   }
